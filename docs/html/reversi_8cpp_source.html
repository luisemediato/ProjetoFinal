136bool Reversi::jogadaEValida() const {

146 return false;

282bool Reversi::capturarPecas(int linha, int coluna, Tabuleiro::Peca peca) {
283 bool capturou = false;
284
285 for (int deltaLinha = -1; deltaLinha <= 1; ++deltaLinha) {
286 for (int deltaColuna = -1; deltaColuna <= 1; ++deltaColuna) {
287 if ((deltaLinha != 0 || deltaColuna != 0) &&
288 direcaoValida(linha, coluna, deltaLinha, deltaColuna, peca)) {
289
290 int x = linha + deltaLinha;
291 int y = coluna + deltaColuna;
292 while (tabuleiro->getPosicao(x, y) != peca) {
293 tabuleiro->setPosicao(x, y, peca);
294 x += deltaLinha;
295 y += deltaColuna;

298 }
299 }
300 }
