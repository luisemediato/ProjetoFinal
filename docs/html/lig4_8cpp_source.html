171bool lig4::verificarVitoria(int linha, int coluna) const {
172 static const int direcoes[4][2] = {{0, 1}, {1, 0}, {1, 1}, {1, -1}};
173 Tabuleiro::Peca peca = tabuleiro->getPosicao(linha, coluna);
174
175 for (const auto& dir : direcoes) {
176 int contagem = 1;
177
178 // Verifica na direção positiva
179 for (int i = 1; i < 4; ++i) {
180 int novaLinha = linha + i * dir[0];
181 int novaColuna = coluna + i * dir[1];
182 if (dentroDosLimites(novaLinha, novaColuna) &&
183 tabuleiro->getPosicao(novaLinha, novaColuna) == peca) {
184 ++contagem;
185 } else {
186 break;
187 }
188 }
189
190 // Verifica na direção negativa
191 for (int i = 1; i < 4; ++i) {
192 int novaLinha = linha - i * dir[0];
193 int novaColuna = coluna - i * dir[1];
194 if (dentroDosLimites(novaLinha, novaColuna) &&
195 tabuleiro->getPosicao(novaLinha, novaColuna) == peca) {
196 ++contagem;
197 } else {
198 break;
199 }
200 }
201
202 // Se encontrou 4 ou mais peças alinhadas, há vitória
203 if (contagem >= 4) {
204 return true;
205 }
206 }
207
208 return false;
209}
